def turn_right():
    for _ in range(3):
        turn_left()

def face_north():
    while not is_facing_north():
        turn_left()
     
face_north()

def escape():
    if front_is_clear() and wall_on_right():
        move()
    elif front_is_clear():
        move()
    elif right_is_clear() and wall_in_front():
        turn_right()
        move()
    elif wall_on_right() or wall_in_front():
        turn_left()
        move()    
    

while not at_goal():
    escape()
